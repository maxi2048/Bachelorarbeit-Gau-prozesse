train_emulator <- function(response, var_list, family, cal_timesteps){
  
  # -------------------------------------------------------
  # Kovariaten extrahieren (nur gültige Punkte)
  # -------------------------------------------------------
  covar <- do.call(cbind.data.frame, lapply(var_list, function(x){
    x[cal_timesteps,,][which(!is.na(response[cal_timesteps,,]))]
  }))
  
  names(covar) <- names(var_list)
  
  # -------------------------------------------------------
  # Modellformel: Tensorprodukt-Smoother
  # -------------------------------------------------------
  # te(...) = flexible nichtlineare Interaktion aller Variablen
  formula <- formula(
    paste0("tf ~ te(", paste(names(covar), collapse = ","), ")")
  )
  
  # -------------------------------------------------------
  # Datensatz für GAM
  # -------------------------------------------------------
  data <- cbind.data.frame(
    tf = response[cal_timesteps,,][which(!is.na(response[cal_timesteps,,]))],
    covar
  )
  
  # -------------------------------------------------------
  # GAM fitten
  # -------------------------------------------------------
  emulator <- mgcv::gam(formula, data = data, family = family)
  
  return(emulator)
}


# =========================================================
# Funktion: Modell evaluieren + Residuen berechnen
# =========================================================

evaluate_emulator <- function(emulator, response, var_list, val_timesteps) {
  
  # -------------------------------------------------------
  # Kovariaten für Validierungsdaten
  # -------------------------------------------------------
  covar <- do.call(cbind.data.frame, lapply(var_list, function(x){
    x[val_timesteps,,][which(!is.na(response[val_timesteps,,]))]
  }))
  
  names(covar) <- names(var_list)
  
  # -------------------------------------------------------
  # Datensatz (nur gültige Punkte)
  # -------------------------------------------------------
  data <- cbind.data.frame(
    tf = response[val_timesteps,,][which(!is.na(response[val_timesteps,,]))],
    covar
  )
  
  # -------------------------------------------------------
  # Vorhersagen des GAM
  # -------------------------------------------------------
  predict_data <- predict(emulator, type = "response", newdata = data)
  
  # =======================================================
  # Residuen 
  # =======================================================
  
  # r = y - y_hat
  residuals_vec <- data$tf - predict_data
  
  # -------------------------------------------------------
  # Vorhersagen zurück ins Raum-Zeit-Array
  # -------------------------------------------------------
  datafield_predict <- response[val_timesteps,,]
  datafield_predict[which(!is.na(datafield_predict))] <- predict_data
  
  # -------------------------------------------------------
  # 🔥 Residuen zurück ins Raum-Zeit-Array
  # -------------------------------------------------------
  residual_field <- response[val_timesteps,,]
  residual_field[which(!is.na(residual_field))] <- residuals_vec
  
  # =======================================================
  # 🔥 Ortsweise Fehlermetriken
  # =======================================================
  
  rmse_field <- array(NA, dim = dim(datafield_predict)[2:3])
  mae_field  <- array(NA, dim = dim(datafield_predict)[2:3])
  mean_residual_field <- array(NA, dim = dim(datafield_predict)[2:3])
  
  for (i in 1:dim(rmse_field)[1]) {
    for (j in 1:dim(rmse_field)[2]) {
      
      obs_ij  <- response[val_timesteps, i, j]
      pred_ij <- datafield_predict[, i, j]
      res_ij  <- residual_field[, i, j]
      
      # RMSE
      rmse_field[i, j] <- suppressWarnings(
        sqrt(mean((obs_ij - pred_ij)^2, na.rm = TRUE))
      )
      
      # MAE
      mae_field[i, j] <- suppressWarnings(
        mean(abs(obs_ij - pred_ij), na.rm = TRUE)
      )
      
      #  Bias
      mean_residual_field[i, j] <- suppressWarnings(
        mean(res_ij, na.rm = TRUE)
      )
    }
  }
  
  # -------------------------------------------------------
  # Rückgabe aller wichtigen Objekte
  # -------------------------------------------------------
  list(
    results = data.frame(
      rmse = sqrt(mean((predict_data - data$tf)^2)),
      mae = mean(abs(predict_data - data$tf)),
      Expl_var = 1 - mean((predict_data - data$tf)^2) / var(data$tf)
    ),
    
    predictions = predict_data,        # für Scatterplots
    residuals_vec = residuals_vec,     # für Histogramm etc.
    residual_field = residual_field,   # Raum-Zeit-Analyse
    
    rmse_field = rmse_field,
    mae_field = mae_field,
    mean_residual_field = mean_residual_field,
    
    data = data
  )
}
